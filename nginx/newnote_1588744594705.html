nginx 轻量级的web代理，处理静态请求，动态请求代理给后端的tomcat，php等



一 下载地址

1 官网： http://nginx.org/en/download.html

指定版本下载

http://nginx.org/download/nginx-1.16.1.tar.gz



2 yum安装

[root@jumpserver yum.repos.d]# rpm -Uvh http://nginx.org/packages/centos/7/noarch/RPMS/nginx-release-centos-7-0.el7.ngx.noarch.rpm

[root@jumpserver ~]# cd /etc/yum.repos.d/

[root@jumpserver yum.repos.d]# cat nginx.repo

# nginx.repo

[nginx]

name=nginx repo

baseurl=http://nginx.org/packages/centos/7/$basearch/

gpgcheck=0

enabled=1



if [ -f /etc/yum.repos.d/nginx.repo ];then

echo "文件存在"

else

touch nginx.repo

fi

cat > /etc/yum.repos.d/nginx.repo << EOF

[nginx-stable]

name=nginx

baseurl=http://nginx.org/packages/centos/$releasever/$basearch/

gpgcheck=1

enabled=1

gpgkey=https://nginx.org/keys/nginx_signing.key

module_hotfixes=true

EOF

依赖包

yum install gcc gcc-c++ pcre pcre-devel zlib zlib-devel openssl openssl-devel
