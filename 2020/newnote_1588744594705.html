yum配置


yum安装

if [ -f /etc/yum.repos.d/nginx.repo ];then

echo "文件存在"

else

touch nginx.repo

fi

cat > /etc/yum.repos.d/nginx.repo << EOF

[nginx-stable]

name=nginx

baseurl=http://nginx.org/packages/centos/$releasever/$basearch/

gpgcheck=1

enabled=1

gpgkey=https://nginx.org/keys/nginx_signing.key

module_hotfixes=true


EOF
